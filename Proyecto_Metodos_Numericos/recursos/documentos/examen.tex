\section*{Examen}

    \item Método Newton--Raphson mejorado:
    \[
    \text{Tol}: 0.001,\quad y=\sin(x),\quad x_0=0.6
    \]
    Respuesta:\rule{3cm}{0.4pt}

        \centering

    \item Método de falsa posición:
    \[
    y = \frac{\sqrt{x^2+2}\,(x+5)}{(x^2-3x-2)^2\,(x+4)^3},\quad a=0,\ b=7,\ \text{Tol}=0.01
    \]
    Respuesta:\rule{3cm}{0.4pt}
    
    \begin{figure}[h]
        \centering
        % tabla para falsa posición
        \includegraphics[width=0.6\textwidth]{tabla2.png}
        \caption{Tabla adicional al aplicar falsa posición.}
    \end{figure}
    \noindent Debido a que $f(0)=113.137$ y $f(7)=168.732$ tienen el mismo signo, el teorema del valor intermedio no
    garantiza ninguna raíz en $[0,7]$, por lo que el método falla.
